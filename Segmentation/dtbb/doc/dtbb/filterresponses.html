0053 % fdimx  number of cols in filter

0059 
0060 padx = pyra.padx;
0061 pady = pyra.pady;
0062 imsize = pyra.imsize;
0063 imarea = imsize(1)*imsize(2);
0064 bboxarea = (bbox(3)-bbox(1)+1)*(bbox(4)-bbox(2)+1);
0065 
0066 % corners for each placement of the filter (in image coordinates)
0067 x1 = ([1:dimx] - padx - 1) * scale + 1;
0068 y1 = ([1:dimy] - pady - 1) * scale + 1;
0069 x2 = x1 + fdimx*scale - 1;
0070 y2 = y1 + fdimy*scale - 1;
0071 if bboxarea / imarea < 0.7
0072   % clip detection window to image boundary only if
0073   % the bbox is less than 70% of the image area
0074   x1 = min(max(x1, 1), imsize(2));
0075   y1 = min(max(y1, 1), imsize(1));
0076   x2 = max(min(x2, imsize(2)), 1);
0077   y2 = max(min(y2, imsize(1)), 1);
0078 end
0079 % intersection of the filter with the bounding box
0080 xx1 = max(x1, bbox(1));
0081 yy1 = max(y1, bbox(2));
0082 xx2 = min(x2, bbox(3));
0083 yy2 = min(y2, bbox(4));
0084 
0085 % e.g., [x1(:) y1(:)] == every upper-left corner
0086 [x1 y1] = meshgrid(x1, y1);
0087 [x2 y2] = meshgrid(x2, y2);
0088 [xx1 yy1] = meshgrid(xx1, yy1);
0089 [xx2 yy2] = meshgrid(xx2, yy2);
0090 % compute width and height of every intersection box
0091 w = xx2(:)-xx1(:)+1;
0092 h = yy2(:)-yy1(:)+1;

0094 % a = area of (possibly clipped) detection windows
0095 a = (x2(:)-x1(:)+1) .* (y2(:)-y1(:)+1);

0097 b = (bbox(3)-bbox(1)+1) * (bbox(4)-bbox(2)+1);
0098 % intersection over union overlap
0099 o = inter ./ (a+b-inter);
